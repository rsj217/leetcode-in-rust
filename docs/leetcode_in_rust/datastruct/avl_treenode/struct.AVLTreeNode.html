pub fn get_height(node: Option<Rc<RefCell<AVLTreeNode>>>) -> i32[src]

get_height 方法用于获取节点的高度，节点不存在的时候，返回0。

Example

use leetcode_in_rust::datastruct::avl_treenode::AVLTreeNode;
use std::rc::Rc;
use std::cell::RefCell;

let root = Some(Rc::new(RefCell::new(AVLTreeNode::new(0))));
let height = AVLTreeNode::get_height(root);
assert_eq!(height, 1);

let root = Some(Rc::new(RefCell::new(AVLTreeNode::new(0))));
let height = AVLTreeNode::get_height(root.as_ref().unwrap().borrow().left.clone());
assert_eq!(height, 0);

pub fn get_bf(node: Option<Rc<RefCell<AVLTreeNode>>>) -> i32[src]

get_bf 方法获取节点的平衡因子（balance factor）。当平衡因子的绝对值大于1的时候，该节点即不平衡。 AVL树要求每一个节点都是平衡的。因此判定每个节点的平衡因子，就可以判定是否是平衡树。

Example

use leetcode_in_rust::datastruct::avl_treenode::AVLTreeNode;
use std::rc::Rc;
use std::cell::RefCell;

let root = Some(Rc::new(RefCell::new(AVLTreeNode::new(0))));
let bf = AVLTreeNode::get_bf(root);
assert_eq!(bf, 0);
